-- DATABASE SETUP MANUALE (CREATION AND POPULATION) --

-- TABLE CREATION --

create table macchinario(
    idMacchina char(10),
    nomeMacchina char(50),
    marca char (50),
    prezzo int(10),
    quantita int(10),
    disponibilita char(2),
    
    primary key (idMacchina)
);

create table utente (
    nome char (50),
    cognome char (50),
    email char (50),
    username char (50),
    password char (255),
    telefono char (255),
    
    primary key (email)
);

create table carrello (
    idCarrello int (255) not null auto_increment,
    emailUtente char (50),
    nomeProdotto char (255),
    prezzoProdotto int (255),
    
    primary key (idCarrello)
);
